Convert decimal numbers to binary. Enjoy! No cheating. No toString

              
function decToBin(d) {
  const arr = [];
  if (d === 0) {
    return '0';
  }
  let i = 0;
  while ( d > 0) {
    arr[i] = d % 2;
    d = d / 2;
    d = Math.floor(d);
    i++;
  }
  return arr.reverse().join('');
}
